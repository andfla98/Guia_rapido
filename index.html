# Hack para organizar downloads automaticamente
import os, shutil
extensoes = {
    "imagens": [".jpg", ".png"],
    "documentos": [".pdf", ".docx"]
}
for arquivo in os.listdir():
    for pasta, exts in extensoes.items():
        if any(arquivo.endswith(ext) for ext in exts):
            os.makedirs(pasta, exist_ok=True)
            shutil.move(arquivo, pasta)
print("✓ Sistema hackeado com sucesso!")
